function convertToHTML() {
let editorContent = tinymce.get('rich-text-editor').getContent();

// Process the HTML
const parser = new DOMParser();
const htmlDoc = parser.parseFromString(editorContent, 'text/html');

// Remove aria-level attributes
const headings = htmlDoc.querySelectorAll('[aria-level]');
for (const heading of headings) {
heading.removeAttribute('aria-level');
}

// Add target="_blank" and rel="noopener noreferrer" to every link
const links = htmlDoc.getElementsByTagName('a');
for (const link of links) {
link.setAttribute('target', '_blank');
link.setAttribute('rel', 'noopener noreferrer');
}

// Process all elements
const allElements = htmlDoc.getElementsByTagName('*');
for (const element of allElements) {
// Remove links from images and set image size
if (element.tagName === 'IMG') {
element.setAttribute('style', 'max-width:100%;height:auto;');
if (element.parentNode.tagName === 'A') {
element.parentNode.outerHTML = element.outerHTML;
}
}
// Remove id, style, dir, and role attributes
element.removeAttribute('id');
element.removeAttribute('style');
element.removeAttribute('dir');
element.removeAttribute('role');
}

// Output plain formatted HTML
document.getElementById('output').value = htmlDoc.body.innerHTML;
}
